void dellmin4(Linklist &L) {
LNode* pre = L, * p = L->next;
LNode* tmp=p,tmppre=pre ;
while (p != NULL) {
if (tmp->data > p->data) {
tmp = p;
tmppre = pre;
}
pre = p;
p = p->next;
}
tmppre->next = tmp->next;
free(tmp);
}

void Linklist_reverse2(Linklist &L) {
LNode *pre = NULL, *p = L->next, * tmp = L;
