int arr[10] = { 1,2,3,4,5,6,1,2,3,4 };
bool reverse(int arr[], int left, int right) {
if (left > right)return false;
int mid = (left+right) / 2;
for (int i = 0; i <= mid - left; i++) {
int tmp = arr[i+left];
arr[i + left] = arr[right - i];
arr[right - i] = tmp;
}
}
bool sqlrev(int arr[],int m, int n) {
reverse(arr, 0, m + n - 1);
reverse(arr,0, n - 1);
reverse(arr, n, m+n - 1);
return true;
}
